project(simulator)
set(CMAKE_CXX_STANDARD 17)

file(GLOB SOURCES
main/*.cpp

../../platform/compiled_proto/proto/*.c
../../platform/spark/device-os/third_party/nanopb/nanopb/*.c
../../brewblox/blox
../../controlbox/src/
)

include_directories(
    ../../platform/simulator
    ../../platform/all
    ../../lib/inc
    ../../lib/cnl/include
    ../../platform/compiled_proto
    ../../brewblox
    ../../platform/spark/device-os/third_party/nanopb/nanopb
    ../../controlbox/src/
    main
    components/lvgl/src
    )


add_subdirectory (components/lvgl)

# add the executable
add_executable(simulator ${SOURCES})
target_link_libraries (simulator LINK_PUBLIC lvgl)
target_link_libraries(simulator PRIVATE Threads::Threads)
